  4. favicon.png → app/assets/favicon.png (web/PWA use, 48×48)

React Native Usage Code
How to use background SVGs in React Native screens
// Import from the central assets index:
import { darkBg, authBg, lightBg, celebrateBg, storeBg } from '../assets';
import { SvgXml } from 'react-native-svg';
import { StyleSheet, View } from 'react-native';

// Use as full-screen background:
const MyScreen = () => (
  <View style={{ flex: 1 }}>
    <SvgXml xml={darkBg} style={StyleSheet.absoluteFillObject} preserveAspectRatio="xMidYMid slice" />
    {/* your content on top */}
  </View>
);

// Or just use LinearGradient (already in your app!):
import { LinearGradient } from 'expo-linear-gradient';
const MyScreen = () => (
  <LinearGradient colors={['#1B3A6B', '#0D7377']} style={{ flex: 1 }}>
    {/* content */}
  </LinearGradient>
);
